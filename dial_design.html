Please ensure it's a valid TTF, OTF, or WOFF file."); } }; // Draw the watch face const drawWatchFace = () => { const canvas = canvasRef.current; if (!canvas) return; const ctx = canvas.getContext('2d'); // Clear and draw background based on high-res size ctx.clearRect(0, 0, RENDER_SIZE, RENDER_SIZE); // Scale the context so our 800px logical math scales up perfectly ctx.save(); ctx.scale(SCALE, SCALE); // Draw Dial Base ctx.fillStyle = faceSettings.bgColor; ctx.beginPath(); ctx.arc(CENTER, CENTER, LOGICAL_SIZE / 2 - 10, 0, 2 * Math.PI); ctx.fill(); // Draw Bezel / Edge ctx.strokeStyle = faceSettings.accentColor; ctx.lineWidth = 15; ctx.stroke(); // Tick Marks ctx.strokeStyle = faceSettings.textColor; for (let i = 0; i < 60; i++) { const isHour = i % 5 === 0; // Skip drawing if the specific mark type is disabled if (isHour && !faceSettings.showHourMarks) continue; if (!isHour && !faceSettings.showMinuteMarks) continue; const angle = (i * Math.PI / 30) - Math.PI / 2; const innerRadius = isHour ? LOGICAL_SIZE / 2 - 40 : LOGICAL_SIZE / 2 - 25; const outerRadius = LOGICAL_SIZE / 2 - 15; ctx.lineWidth = isHour ? 4 : 2; ctx.beginPath(); ctx.moveTo(CENTER + innerRadius * Math.cos(angle), CENTER + innerRadius * Math.sin(angle)); ctx.lineTo(CENTER + outerRadius * Math.cos(angle), CENTER + outerRadius * Math.sin(angle)); ctx.stroke(); } // Draw Hour Markings (Roman Numerals) ctx.fillStyle = faceSettings.textColor; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.font = `${hourSettings.size}px "${hourSettings.font}"`; ctx.letterSpacing = `${hourSettings.letterSpacing}px`; const numerals = ['I', 'II', 'III', hourSettings.useWatchmakersFour ? 'IIII' : 'IV', 'V', 'VI', 'VII', 'VIII', 'IX', 'X', 'XI', 'XII']; numerals.forEach((num, index) => { const hour = index + 1; const angle = (hour * Math.PI / 6) - Math.PI / 2; const x = CENTER + hourSettings.radius * Math.cos(angle); const y = CENTER + hourSettings.radius * Math.sin(angle); // Canvas textAlign 'center' includes the trailing letter spacing in its width calculation, // which offsets the visual center. We compensate by shifting by half the spacing distance. const spacingCorrection = hourSettings.letterSpacing / 2; if (hourSettings.rotateText) { ctx.save(); ctx.translate(x, y); // Rotate text radially (bottom of text points to the center) ctx.rotate(angle + Math.PI / 2); ctx.fillText(num, spacingCorrection, 0); ctx.restore(); } else { ctx.fillText(num, x + spacingCorrection, y); } }); // Reset letter spacing for subsequent text ctx.letterSpacing = '0px'; // Draw Brand Text ctx.font = `bold ${brandSettings.size}px "${brandSettings.font}"`; ctx.fillText(brandSettings.text, CENTER, brandSettings.yPos); // Draw Model Text ctx.font = `${modelSettings.size}px "${modelSettings.font}"`; ctx.fillText(modelSettings.text, CENTER, modelSettings.yPos); // Restore context to prevent compound scaling on next frame ctx.restore(); }; // Redraw when settings change useEffect(() => { // Ensure fonts are ready before drawing document.fonts.ready.then(() => { drawWatchFace(); }); }, [hourSettings, brandSettings, modelSettings, faceSettings]); // Export Canvas const downloadImage = () => { const canvas = canvasRef.current; if (!canvas) return; const dataUrl = canvas.toDataURL('image/png'); const link = document.createElement('a'); link.download = 'custom-watch-face.png'; link.href = dataUrl; link.click(); }; // Reusable File Input Component const FontUploader = ({ section, currentFontName }) => (
